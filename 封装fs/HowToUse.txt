var fs=require('./fseasyFZ')
1.读取文件(read file)
   fs.readFile(url,function(err,data){})
   url:读取文件路径
   function(err,data){}
   err:是否出错;
   data:读取到的内容
2.写入文件(wirte file)
   fs.wirteFile(url,text,function(err,data){})
   text:写入的内容
3.复制文件
   fs.copyFile(url1,url2,function(err,data){})
   url1:被复制的文件路径
   url2:要复制到的路径
4.删除文件
   fs.deleteFile(url,function(err){})
5.修改文件名字
   fs.rename(url,newname,function(err){})
6.判断是否存在及文件类型
   fs.existFile(url)
7.把内容追加到末尾
   fs.appendFile(url,text,function(err){})
